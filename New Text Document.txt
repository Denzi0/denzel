DELIMITER //
CREATE FUNCTION NO_OF_YEARS (DATE1 DATE)
RETURNS INT DETERMINISTIC
BEGIN
	DECLARE DATE2 DATE;
	SELECT CURRENT.DATE() INTO DATE2;
	RETURN YEAR(DATE2) - YEAR(DATE1);
end //
